polspline_stats = \(x, ...) {
    fit = polspline::logspline(x)
    return(list(
      density = \(k, ...) polspline::dlogspline(k, fit),
      quantile = \(k, ...) polspline::qlogspline(k, fit)
    ))
}

normal_stat = \(x, ...) {
  mu = mean(x)
  sigma = stats::sd(x)
  return(\(k, ...) stats::dnorm(k, mu, sigma))
}
